function getSignificanceLabel(sig: string) {
  if (sig === "***") return "p<0.001";
  if (sig === "**") return "p<0.01";
  if (sig === "*") return "p<0.05";
  return "n.s.";
}

function getSignificanceColor(sig: string) {
  if (sig === "***")
    return "bg-emerald-500/15 text-emerald-400 border-emerald-500/30";
  if (sig === "**") return "bg-sky-500/15 text-sky-400 border-sky-500/30";
  if (sig === "*")
    return "bg-amber-500/15 text-amber-400 border-amber-500/30";
  return "bg-secondary text-muted-foreground border-border";
}

function getPlainEnglish(variable: string, beta: number): string {
  const dir = beta > 0 ? "higher" : "lower";
  const change = beta > 0 ? "increase" : "decrease";
  const pp = Math.abs(beta * 100).toFixed(1);

  switch (variable) {
    case "HIPRating":
      return `For each 1-point increase in HIP score, returns ${change} by ${pp} percentage points`;
    case "ManagementPractices":
      return `Better management practices are associated with ${pp}pp ${dir} returns`;
    case "Accountability":
      return `Higher accountability scores are associated with ${pp}pp ${dir} returns`;
    case "Wealth":
      return `Higher Wealth pillar scores are associated with ${pp}pp ${dir} returns`;
    case "Health":
      return `Higher Health pillar scores are associated with ${pp}pp ${dir} returns`;
    case "Earth":
      return `Higher Earth pillar scores are associated with ${pp}pp ${dir} returns`;
    case "Equality":
      return `Higher Equality pillar scores are associated with ${pp}pp ${dir} returns`;
    case "Trust":
      return `Higher Trust pillar scores are associated with ${pp}pp ${dir} returns`;
    case "Intercept":
      return `Baseline return when all predictors are zero`;
    default:
      return `Each unit increase is associated with ${pp}pp ${dir} returns`;
  }
}

"use client";
import { useEffect, useRef, useState, useCallback } from "react";

export default function AnimatedCounter({ end, duration = 2000, decimals = 0 }: { end: number; duration?: number; decimals?: number }) {
  const [count, setCount] = useState(0);
  const ref = useRef<HTMLSpanElement>(null);
  const hasAnimated = useRef(false);

  const runAnimation = useCallback(() => {
    if (hasAnimated.current) return;
    hasAnimated.current = true;
    const startTime = performance.now();
    const animate = (currentTime: number) => {
      const elapsed = currentTime - startTime;
      const progress = Math.min(elapsed / duration, 1);
      const eased = 1 - Math.pow(1 - progress, 3);
      setCount(eased * end);
      if (progress < 1) requestAnimationFrame(animate);
    };
    requestAnimationFrame(animate);
  }, [end, duration]);

  useEffect(() => {
    const el = ref.current;
    if (!el) return;

    // Fallback: if already in viewport on mount, animate immediately
    const rect = el.getBoundingClientRect();
    if (rect.top < window.innerHeight && rect.bottom > 0) {
      runAnimation();
      return;
    }

    const observer = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting) {
        runAnimation();
        observer.disconnect();
      }
    }, { threshold: 0.1 });
    observer.observe(el);
    return () => observer.disconnect();
  }, [runAnimation]);

  return <span ref={ref} className="font-mono tabular-nums">{decimals > 0 ? count.toFixed(decimals) : Math.round(count)}</span>;
}

"use client";
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer, ReferenceLine, ReferenceArea } from "recharts";
import { annualReturns } from "@/lib/data";

function CustomTooltip({ active, payload, label }: { active?: boolean; payload?: Array<{ color: string; name: string; value: number }>; label?: string }) {
  if (!active || !payload) return null;
  return (
    <div className="rounded-lg border border-border bg-card p-3 shadow-xl">
      <p className="mb-2 text-sm font-semibold text-foreground">{label}</p>
      {payload.map((entry, i) => (
        <div key={i} className="flex items-center gap-2 text-xs">
          <div className="h-2.5 w-2.5 rounded-full" style={{ backgroundColor: entry.color }} />
          <span className="text-muted-foreground">{entry.name}:</span>
          <span className="font-mono font-semibold" style={{ color: entry.value >= 0 ? "#10b981" : "#ef4444" }}>
            {(entry.value * 100).toFixed(1)}%
          </span>
        </div>
      ))}
    </div>
  );
}

export default function PerformanceChart({ showCumulative = false }: { showCumulative?: boolean }) {
  const data = showCumulative
    ? annualReturns.reduce<{ year: number; high: number; medium: number; low: number }[]>((acc, ret) => {
        const prev = acc.length > 0 ? acc[acc.length - 1] : { high: 1, medium: 1, low: 1 };
        acc.push({ year: ret.year, high: +(prev.high * (1 + ret.high) - 1).toFixed(4), medium: +(prev.medium * (1 + ret.medium) - 1).toFixed(4), low: +(prev.low * (1 + ret.low) - 1).toFixed(4) });
        return acc;
      }, [])
    : annualReturns;

  return (
    <ResponsiveContainer width="100%" height={480}>
      <LineChart data={data} margin={{ top: 10, right: 30, left: 10, bottom: 10 }}>
        <CartesianGrid strokeDasharray="3 3" stroke="rgba(255,255,255,0.06)" />
        <XAxis dataKey="year" stroke="#6b7280" fontSize={12} tickLine={false} />
        <YAxis stroke="#6b7280" fontSize={12} tickLine={false} tickFormatter={(v: number) => `${(v * 100).toFixed(0)}%`} />
        <Tooltip content={<CustomTooltip />} />
        <Legend wrapperStyle={{ fontSize: "12px", paddingTop: "10px" }} iconType="circle" iconSize={8} />
        {!showCumulative && <ReferenceLine y={0} stroke="rgba(255,255,255,0.2)" strokeDasharray="4 4" />}
        <ReferenceArea x1={2021} x2={2022} fill="rgba(16, 185, 129, 0.05)" stroke="rgba(16, 185, 129, 0.2)" strokeDasharray="3 3" label={{ value: "Crossover", position: "insideTop", fill: "#6b7280", fontSize: 10 }} />
        <Line type="monotone" dataKey="high" stroke="#10b981" strokeWidth={2.5} dot={{ r: 4, fill: "#10b981" }} activeDot={{ r: 6 }} name="High HIP" />
        <Line type="monotone" dataKey="medium" stroke="#0ea5e9" strokeWidth={2.5} dot={{ r: 4, fill: "#0ea5e9" }} activeDot={{ r: 6 }} name="Medium HIP" />
        <Line type="monotone" dataKey="low" stroke="#f59e0b" strokeWidth={2.5} dot={{ r: 4, fill: "#f59e0b" }} activeDot={{ r: 6 }} name="Low HIP" />
      </LineChart>

"use client";

import { createContext, useContext, useState, ReactNode } from "react";

interface SidebarState {
  collapsed: boolean;
  setCollapsed: (v: boolean) => void;
  mobileOpen: boolean;
  setMobileOpen: (v: boolean) => void;
}

const SidebarContext = createContext<SidebarState | null>(null);

export function SidebarProvider({ children }: { children: ReactNode }) {
  const [collapsed, setCollapsed] = useState(false);
  const [mobileOpen, setMobileOpen] = useState(false);

  return (
    <SidebarContext.Provider value={{ collapsed, setCollapsed, mobileOpen, setMobileOpen }}>
      {children}
    </SidebarContext.Provider>
  );
}

export function useSidebar() {
  const ctx = useContext(SidebarContext);
  if (!ctx) throw new Error("useSidebar must be used within SidebarProvider");
  return ctx;
}
